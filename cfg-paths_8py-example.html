42 # Avoid duplicated effort by caching the uuids of Blocks from
43 # which we have know the target is unreachable.
44 unreachability_cache = set([])

46 def print_paths(blocks, edges, target_id, vertex_id, visited=set(), path=[]):
47  """Depth-first search of a graph, printing all paths between two vertices."""
48  rv = 0
49  visited.add(vertex_id)
50  path.append(blocks[vertex_id].address)
51 
52  # At target, print the path
53  if vertex_id == target_id:
54  print(', '.join('0x%.8x' % s for s in path))
55  rv= 1
56 
57  # Otherwise, check all outgoing edges from this vertex
58  else:
59  for v in edges[vertex_id]:
60  if v not in visited and v not in unreachability_cache:
61  rv += print_paths(blocks, edges, target_id, v, visited, path)
62 
63  path.pop()
64  # Unmark the node so it can be visited again in other paths
65  visited.discard(vertex_id)
66  if rv==0:
67  unreachability_cache.add(vertex_id)
68  return rv
