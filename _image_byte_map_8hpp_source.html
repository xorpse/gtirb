312  auto DestSpan = as_writeable_bytes(gsl::make_span(&Result, 1));
313  // Assign this to a variable so it isn't destroyed before we copy
314  // from it (because gsl::span is non-owning).
315  const_range Data = this->data(A, DestSpan.size_bytes());
316  std::copy(Data.begin(), Data.end(), DestSpan.begin());
317  return Data.begin() != Data.end();
318  }
