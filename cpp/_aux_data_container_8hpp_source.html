105  template <typename Schema> const typename Schema::Type* getAuxData() const {
106  auto Found = this->AuxDatas.find(Schema::Name);
107 
108  if (Found == this->AuxDatas.end())
109  return nullptr;
110 
111  AuxData& AD = *(Found->second);
112 
113  assert(AD.getApiTypeId() != AuxData::UNREGISTERED_API_TYPE_ID &&
114  "Attempting to retrieve AuxData with an unregistered type.");
115 
116  assert(AD.getApiTypeId() == AuxDataImpl<Schema>::staticGetApiTypeId() &&
117  "Attempting to retrieve AuxData with incorrect type.");
118  auto& ADI = static_cast<AuxDataImpl<Schema>&>(AD);
119  return ADI.get();
120  }

240  template <
241  class MessageType,
242  class = std::enable_if_t<message_has_aux_data_container_v<MessageType>>>
243  void fromProtobuf(const MessageType& Message) {
244  this->AuxDatas.clear();
245  for (const auto& M : Message.aux_data()) {
246  std::unique_ptr<AuxData> Val;
247  std::string Key = M.first;
248 
249  // See if the name for this AuxData is registered.
250  if (const auto* ADT = lookupAuxDataType(Key)) {
251  Val = ADT->fromProtobuf(M.second);
252  }
253 
254  // If it wasn't registered or was registered with an incompatible
255  // type wrt the serialized object, unserialized as just the
256  // un-typed raw data.
257  if (!Val) {
258  Val = std::make_unique<AuxData>();
259  AuxData::fromProtobuf(*Val, M.second);
260  }
261  this->AuxDatas.insert(std::make_pair(Key, std::move(Val)));
262  }
264  }
