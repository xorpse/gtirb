72 // Helper for constructing the CFG type. The graph property needs to refer to
73 // the graph's vertex_descriptor type. This is accessible via
74 // boost::adjacency_list_traits, but requires keeping the template parameters
75 // for boost::adjacency_list and boost::adjacency_list_traits in sync. This
76 // helper ensures the relevant parameters are the same for both.
77 template <class OutEdgeListS = boost::vecS, class VertexListS = boost::vecS,
78  class DirectedS = boost::directedS, class EdgeListS = boost::listS>
79 struct CfgBuilder {
80  using vertex_descriptor = typename boost::adjacency_list_traits<
81  OutEdgeListS, VertexListS, DirectedS, EdgeListS>::vertex_descriptor;
82  using type = boost::adjacency_list<
83  OutEdgeListS, VertexListS, DirectedS,
84  // Vertices are CfgNodes.
85  CfgNode*,
86  // Edges have labels.
87  EdgeLabel,
88  // The graph keeps track of vertex descriptors for
89  // each node.
90  std::unordered_map<const CfgNode*, vertex_descriptor>, EdgeListS>;
91 };
