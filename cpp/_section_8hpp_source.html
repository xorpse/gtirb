297  std::optional<Addr> getAddress() const {
298  if (ByteIntervals.empty()) {
299  return std::nullopt;

322  std::optional<uint64_t> getSize() const {
323  if (ByteIntervals.empty()) {
324  return std::nullopt;
325  }
326 
327  Addr LowAddr{std::numeric_limits<Addr::value_type>::max()};
328  Addr HighAddr{0};
329 
330  for (const auto* Interval : ByteIntervals) {
331  if (auto A = Interval->getAddress()) {
332  LowAddr = std::min(LowAddr, *A);
333  HighAddr = std::max(HighAddr, *A + Interval->getSize());
334  } else {
335  return std::nullopt;
336  }
337  }
338 
339  return static_cast<uint64_t>(HighAddr - LowAddr);
340  }
